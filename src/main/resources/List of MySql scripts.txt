CREATE  TABLE `motorpoll`.`Users` (
  `idUsers` INT NOT NULL AUTO_INCREMENT ,
  `Name_` VARCHAR(45) NULL ,
  `Surname` VARCHAR(45) NULL ,
  `Password` VARCHAR(30) NULL,
  `Role` ENUM('A','U') NULL ,
  `Activity` ENUM('free','busy') NULL ,
  PRIMARY KEY (`idUsers`) ,
  UNIQUE INDEX `idUsers_UNIQUE` (`idUsers` ASC) );


CREATE  TABLE `motorpoll`.`Bus` (
  `idBus` INT NOT NULL AUTO_INCREMENT ,
  `BusName` VARCHAR(45) NULL ,
  `Activity` ENUM('free','busy') NULL ,
  PRIMARY KEY (`idBus`) ,
  UNIQUE INDEX `idBus_UNIQUE` (`idBus` ASC) );


CREATE  TABLE `motorpoll`.`Route` (
  `idRoute` INT NOT NULL AUTO_INCREMENT ,
  `Name` VARCHAR(60) NULL ,
  PRIMARY KEY (`idRoute`) ,
  UNIQUE INDEX `idRoute_UNIQUE` (`idRoute` ASC) );


CREATE  TABLE `motorpoll`.`Work` (
  `idWork` INT NOT NULL AUTO_INCREMENT ,
  `Users_id` INT NULL ,
  `Route_id` INT NULL ,
  `Bus_id` INT NULL ,
  `Accepted` ENUM('Y','N') NULL ,
  PRIMARY KEY (`idWork`) ,
  UNIQUE INDEX `idWork_UNIQUE` (`idWork` ASC) );
